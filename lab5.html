(printf "\nEnter d: ")
(define d (read))

(define nokc 0)
(define nokz 0)

(define (main)
  (displayAll "\nAnswer: " (calculate-lcm a b c d))
  )

(define (calculate-lcm a b c d) ;To find lcm for rational numbers
  (begin
    (set! nokz (lcm b d))
    (set! nokc (lcm (* a (/ (lcm b d) b)) (* c (/ (lcm b d) d))))
    (/ nokc nokz)
    )
  )
  

(main)
